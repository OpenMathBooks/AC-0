\item What is the phase shift of $f(x) = \dfrac{1}{5} \tan \left (2x + \dfrac{\pi}{2} \right )$?
	
	\begin{enumerate}
		\item $2\pi$
		\item $\pi$
		\item $\dfrac{\pi}{2}$
		\item $\dfrac{\pi}{4}$
		\item $-2\pi$
		\item $-\pi$
		\item $-\dfrac{\pi}{2}$
		\item $-\dfrac{\pi}{4}$
	\end{enumerate}
