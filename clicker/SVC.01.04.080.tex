\item Solve for $x$ if $y = e + 2^x$

\begin{enumerate}
\item $x = \frac{\ln y - 1}{\ln 2}$
\item $x = \frac{\ln (y - 1)}{\ln 2}$
\item $x = \frac{\ln y }{\ln 2} -1$
\item $x = \frac{\ln (y - e)}{\ln 2}$
\end{enumerate}
