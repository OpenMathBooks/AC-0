\item Solve for $x$ if $8y = 3e^x$.

\begin{enumerate}
\item $x = \ln 8 + \ln 3 + \ln y$
\item $x = \ln 3 - \ln 8 + \ln y$
\item $x = \ln 8 + \ln y - \ln 3$
\item $x = \ln 3 - \ln 8 - \ln y$
\end{enumerate}
