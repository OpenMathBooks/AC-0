\item What is the equation of the function shown in the graph?

\begin{minipage}{0.4\columnwidth}
    \begin{enumerate}
        \item $y = 3 \sin (2x)+2$
        \item $y = 3\cos(2x)+2$
        \item $y=3\sin(\pi x)+2$
        \item $y = 3 \cos (\pi x) +2$
        \item $y=3 \sin (\frac{1}{\pi} x) +2$
        \item $y = 3 \cos((\frac{1}{\pi} x)+2)$
    \end{enumerate}
\end{minipage}
