\item If $P = f(t) = 3 + 4t$, find $f^{-1}(P)$.

\begin{enumerate}
\item $f^{-1}(P) = 3+4P$
\item $f^{-1}(P) = \frac{P-3}{4}$
\item $f^{-1}(P) = \frac{P-4}{3}$
\item $f^{-1}(P) = 4(P+3)$
\item $f^{-1}(P) = \frac{P+3}{4}$
\end{enumerate}
