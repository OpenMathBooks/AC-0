\begin{activity}\label{A:0.4.2}
    Solve each of the following equations for $t$, and verify your answers using a calculator.
	\ba
		\item $\ln t=4$
		\item $\ln(t+3)=4$
		\item $\ln(t+3)=\ln4$
		\item $\ln(t+3)+\ln(t)=\ln4$
		\item $e^{t}=4$
		\item $e^{t+3}=4$
		\item $2e^{t+3}=4$
		\item $2e^{3t+2}=3e^{t-1}$
	\ea

\end{activity}\aftera
